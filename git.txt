git checkout -b <分支名> 新建并切换到新分支
git branch -d <分支名> 删除分支
git push origin --delete <分支名> 删除远程分支
git remote -v 查看远程仓库名和url
git checkout -- [文件] 丢弃指定文件修改恢复到以前状态
git checkout -- . 丢弃所有修改恢复到以前状态
git reset HEAD~1 -hard 重置到上一次提交 git undo
git remote prune origin 清理无效的远程分支引用
git config user.name "新用户名" 更换用户名
